cmake_minimum_required(VERSION 2.8.3)
project(robotis_controller)

set(CMAKE_CXX_FLAGS "-std=c++11 ${CMAKE_CXX_FLAGS}")

find_package(catkin REQUIRED COMPONENTS
  roscpp
  roslib
  sensor_msgs
  std_msgs
  robotis_device
  robotis_controller_msgs
  robotis_framework_common
  cmake_modules
  dynamixel_sdk
  dynamic_reconfigure
)

generate_dynamic_reconfigure_options(
  cfg/Offsets.cfg
 )

catkin_package(
  INCLUDE_DIRS include
  LIBRARIES robotis_controller
  CATKIN_DEPENDS roscpp roslib sensor_msgs std_msgs dynamic_reconfigure
#  DEPENDS system_lib
)

include_directories(
  include
  ${catkin_INCLUDE_DIRS}
)

add_library(robotis_controller
  src/robotis_controller/robotis_controller.cpp
)

add_dependencies(robotis_controller ${robotis_controller_EXPORTED_TARGETS})

target_link_libraries(robotis_controller
  yaml-cpp
  ${catkin_LIBRARIES}
)
